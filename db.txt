CREATE DATABASE loja;
USE loja;
CREATE TABLE produto(
codProd INT PRIMARY KEY AUTO_INCREMENT,
descProd VARCHAR(90) NOT NULL,
valorProd DECIMAL(5,2) NOT NULL,
figuraProd INT NOT NULL
);
/*futuramente figura sera blob*/
CREATE TABLE cliente(
codCliente INT PRIMARY KEY AUTO_INCREMENT,
nomeCliente VARCHAR(90) NOT NULL,
cpfCliente VARCHAR(11) NOT NULL,
cidadeCliente VARCHAR(90) NOT NULL,
generoCliente TINYINT NOT NULL
);
CREATE TABLE mensagem(
codMens INT PRIMARY KEY AUTO_INCREMENT,
assuntoMens VARCHAR(90),
obsMens VARCHAR(90) NOT NULL,
dataMens DATE NOT NULL
);
